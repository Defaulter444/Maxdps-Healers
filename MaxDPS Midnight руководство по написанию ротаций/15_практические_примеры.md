15. Практические примеры
Простой ДПС-листинг (Воин Оружия)
# Поддерживать Battle Shout
actions+=battle_shout,if=buff.battle_shout.down
# Execute на низком здоровье цели
actions+=execute,if=target.health.pct<20
# Mortal Strike по готовности
actions+=mortal_strike,if=cooldown.mortal_strike.up
# АоЕ при нескольких целях
actions+=whirlwind,if=active_enemies>=3&rage>=40
# Наполнитель
actions+=slam,if=rage>30
Ротация со списками (Воин Оружия)
# Основной список
actions+=battle_shout,if=buff.battle_shout.down
actions+=call_action_list,name=aoe,if=active_enemies>=3
actions+=call_action_list,name=st,if=active_enemies<3
# AOE список
actions.aoe+=whirlwind,if=rage>=40
actions.aoe+=cleave
# ST список
actions.st+=execute,if=target.health.pct<20
actions.st+=mortal_strike,if=cooldown.mortal_strike.up
actions.st+=slam
Использование переменных
# Один раз считаем, пора ли готовить бурст
actions+=variable,name=need_burst,value=cooldown.avatar.remains<5
# Используем переменную в нескольких местах
actions+=avatar,if=variable.need_burst
actions+=overpower,if=variable.need_burst|cooldown.mortal_strike.remains<1
